[docs]    def run_test(self):
        """This method runs the curve fitting test on data.
        
        it runs an n x m factorial experiment on various slope
        and gaussian noise ranges and renders the results as a
        2D or 3D plot.
        
        Parameter:
            None
            
        Returns:
            None
        """
        self.results = []
        self.slopes = np.linspace(self.txtSlopeMin.value(), self.txtSlopeMax.value(), self.txtSlopeNum.value())
        self.stds = np.linspace(self.txtStdMin.value(), self.txtStdMax.value(), self.txtStdNum.value())
        
        for i in range(len(self.slopes)):
            for j in range(len(self.stds)):
                title = 'slope_'+str(self.slopes[i])+' std_'+str(self.stds[j])
                self.results.append(Data(title, m = self.slopes[i], std = self.stds[j], num_samples = self.txtSamples.value(), x_factor = self.txtXscale.value(), files = self.cbFiles.isChecked(), plots = self.cbPlots.isChecked()))
        
        self.res = [np.round(i.data['gradient']-i.data['grad_pred'], 3) for i in self.results]
        self.grad_pred = [np.round(i.data['grad_pred'],2) for i in self.results]
        self.res = np.reshape(self.res,(len(self.slopes), len(self.stds)))
        
        self.tblData.clear()
        
        self.tblData.setColumnCount(len(self.stds))
        self.tblData.setRowCount(len(self.slopes))
        
        HLabels = ''
        for i in self.slopes:
            HLabels += str(np.round(i,2))+','
        HLabels.rstrip()
        
        VLabels = ''
        for i in self.stds:
            VLabels += str(np.round(i,2))+','
        VLabels.rstrip()
        
        self.tblData.setHorizontalHeaderLabels(qt.QString(VLabels).split(','))
        self.tblData.setVerticalHeaderLabels(qt.QString(HLabels).split(','))
        
        for i in range(len(self.slopes)):
            for j in range(len(self.stds)):
                self.tblData.setItem(i,j,gui.QTableWidgetItem(str(self.res[i,j]))) 
        
        self.plot()

[docs]class Data(object):
    """This is the test data generation class
    
    It is used to generate arbitrary sample data with specified
    properties given by the user.
    
    Parameter:
            name (str): the name associated with the generated data
            m (float): the gradient of the generated data. Default value is 5
            std (float): the standard deviation of the generated data. Default value is 0.4
            num_samples (int): the sample size of the generated data. Default value is 100
            x_factor (float): the x-scaling factor of the generated x-data. Default value is 5
            files (bool): variable specifies whether the data files should be saved. it saves if it's value is true. Default value is False
            plots (bool): variable specifies whether the plots should be saved. it saves if it's value is true. Default value is False
            
    """
    
    def __init__(self, name, m = 5, std = 0.4, num_samples = 100, x_factor = 1, files = False, plots = False):
        """The method that initializes the data class

        Parameter:
            name (str): the name associated with the generated data
            m (float): the gradient of the generated data. Default value is 5
            std (float): the standard deviation of the generated data. Default value is 0.4
            num_samples (int): the sample size of the generated data. Default value is 100
            x_factor (float): the x-scaling factor of the generated x-data. Default value is 5
            files (bool): variable specifies whether the data files should be saved. it saves if it's value is true. Default value is False
            plots (bool): variable specifies whether the plots should be saved. it saves if it's value is true. Default value is False
            
        Returns:
            None
        """
        self.name = name
        self.m = m #mean
        self.std = std #standard deviation
        self.num = num_samples #number of data samples
        self.x_factor = x_factor #multiplier for x data values
        self.gradient = None
        self.intercept = None
        self.y_predict = None
        self.figure = None
        
        #noise = np.random.normal(0, self.std, size=self.num)
        noise = [random.gauss(0, self.std) for i in range(self.num)]#generation of gaussian noise
        
        self.x_data = (np.array(range(self.num)))*self.x_factor #generate x-data values
        Y = (self.m*self.x_data) + noise #generate y-data values
        self.y_data = self.preprocess_list_line(Y.reshape(self.num,1))
        
        self.evaluate()# predict gradient
        
        self.error = self.m - self.gradient
        if plots:
            self.save_plot()
        if files:
            self.write_xls()
        
        self.data = dict() #initialize a dictionary data object to hold/store generated data information
        self.data['name'] = name
        self.data['gradient'] = self.m
        self.data['y_data'] = self.y_data
        self.data['num'] = self.num
        self.data['std'] = self.std
        self.data['y_pred'] = self.y_predict
        self.data['grad_pred'] = self.gradient
        self.data['error'] = self.error
        
    def __str__(self):
        """This method specifies the string representation of data class
        
        Parameter:
            None
            
        Returns:
            None
        """
        
        return str(self.data)

        """This method is used to save plots as images in the image directory

        """
        self.figure,ax = plt.subplots(1,1)

[docs]    def read_xls(self, name):
        """This method reads an excel file from a specified path
        
        Parameter:
            name (str): specifies the path and name of the excel file to be read
            
        Returns:
            None
        """
        book = load_workbook(name)
        sheet = book.active
        data = []
        ind = 1
        while True:
            y = sheet['A' +str(ind)].value
            ind+=1
            if y == None:
                break
            else:
                data.append(y)
        return data

        """return a list of Y data from read file"""
        """This method is used to format y-data
        
        Parameter:
            data (str): y-data string read from file
            
        Returns:
            list: return a list of y-data from read file
        """
        data = str(data)
        res = data.replace('[','')
        res = res.replace(']', '')
        res = res.replace(' ', '')
        res = res.split('\n')
        bn = []
        [bn.append(float(i)) for i in res]
        return bn
